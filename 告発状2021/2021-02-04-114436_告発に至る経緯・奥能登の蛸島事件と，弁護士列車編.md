#!/bin/bash

if [ $# -ne 1 ]; then
  echo "指定された引数は$#個です。" 1>&2
  echo "実行するにはTwilogcsvのファイル名を１つ引数で指定。" 1>&2
  exit 1
fi

cd ~/twilog
prev_file=$1
user=${prev_file%%[0-9]*\.csv}
pp=`sed -nE '1s/^([0-9]{4}-[0-9]{2}-[0-9]{2} [0-9]{2}:[0-9]{2}:[0-9]{2}).+/\1/p' $prev_file`; echo $pp
next_file=`basename ~/Downloads/${prev_file%%[0-9]*\.csv}*.csv.gz`
mv ~/Downloads/$next_file ~/twilog/tmp.csv.gz
new_file=${next_file%%.gz}

next_file=tmp.csv
gunzip ${next_file}.gz

ruby -pne '$_.gsub!(/\n/, " ")' $next_file | ruby -pne '$_.gsub!(/" "/, "\"\n\"")'|sed '$s/ $/\n/' > a && mv a $next_file
sed -i -E "s/\"([0-9]+)\",\"([0-9]{2})([0-9]{2})([0-9]{2}) ([0-9]{2})([0-9]{2})([0-9]{2})\",\"(.+)\"$/20\2-\3-\4 \5:\6:\7 \"\8\" https:\/\/twitter.com\/${user}\/status/\/\1/" $next_file
echo "new $next_file"
echo "prev $prev_file"
sed -i  '1i XXX' $prev_file && sed -n "1,$((`sed -n "/${pp}/=" ${next_file}` - 1))p" $next_file | sed -i '1r /dev/stdin' $prev_file && sed -i '/^XXX$/d' $prev_file
mv tmp.csv $new_file

if [ $prev_file != $new_file ]; then
  rm $prev_file
fi

sed -n "1,$((`sed -n "/${pp}/=" ${new_file}` - 1))p" $new_file | wc -l 
 echo "${new_file}を更新しました。"

　確認すると昨夜22時14分からの中断となっていましたが，夕方の18時台には中断をしていたような感覚でいました。昨夜もいろいろと発見があったり，調べることがあったのですが，全てを取り上げる暇はありません。

　特に大きなニュースに感じたのが，大津地検の女性検事正就任会見でした。裁判官の場合は年度かわり以外の転任をみることが多いのですが，検察庁に関しては年度替わりでの転任がきっちりしているものと，金沢地方検察庁の担当者の交代の経験からも思い込みがありました。

　検事の情報としては見たことがないのですが，裁判官の場合は経歴を掲載したページがあり，そこに転任の時期が年月になっていて，4月以外のものを多く見かけていたということです。

　小倉秀夫弁護士のアカウントの反応は今回も未確認ですが，私のアカウントをミュートしている可能性が高く，他のアカウントから伝えられない限りは知ることがないと推定されます。

　小倉秀夫弁護士は古風な一面が感じられ，伝統的な弁護士の精神，気質がよく現れていると感じることがあります。別の見方をすれば，過去の弁護士の功績の受け売りなのですが，一昔前の価値観を感じさせるものが残っています。

　その弁護士の過去の功績を代表する金字塔みたいなのが蛸島事件になるのですが，ネットで知ることの出来た情報というのはほんの一部で，実態とはかけ離れた一面を感じました。共通するのは，弁護士脳の社会汚染と危険性という問題です。

　ただ，昭和40年代前半とは時代背景もまるで異なるので，いちがいに決めつけるのもどうかと思います。大きな刑事裁判になった事件というのは，私の知る限り奥能登では他にはないのですが，昭和40年代前半というのは1歳から5歳頃のことで，ある程度の記憶は残っています。

　とにかく，石川県奥能登でおきた蛸島学童殺人事件裁判記録はすごい本でした。平成5年辺りから刑事裁判には強い関心を持ち調べてきましたが，それまで前提としてきた基準が前提から覆されたような衝撃です。

　これから，その石川県奥能登でおきた蛸島学童殺人事件裁判記録の紹介をしていきますが，現在の，そして未来に繋がるモトケンこと矢部善朗弁護士（京都弁護士会），小倉秀夫弁護士，深澤諭史弁護士などの弁護士らの現状，生態系を交えて考察を深めて頂ければと思います。

　小倉秀夫弁護士は余り裕福な家ではなかったらしく，そういうニュアンスのツイートを時折り見かけてきたのですが，特に印象的だったエピソードの1つに，子供の頃は図書館に入り浸りだったというものがありました。
